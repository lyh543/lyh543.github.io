# 最原始的求法

int comb(int n, int m)
{
    int p = 1, q = 1;
    for (int i = 1; i <= m; i++)
    {
        p = p * (n - i + 1);
        q = q * i;
    }
    if (p % q == 0) p /= q;
    return p / q;
}

{
    return a >= b ? fac[a] * inv(fac[b]) % m * inv(fac[a-b]) % m : 0;
